----
Create the tekton-pvc-name using the marcsc storage class
----

----
kind: PersistentVolumeClaim
apiVersion: v1
metadata:
  name: tekton-pvc-name
spec:
  accessModes:
  - ReadWriteOnce
  volumeMode: Filesystem
  resources:
    requests:
      storage: 100Gi
  storageClassName: marcsc
----
